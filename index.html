2. SQL 查询示例（共 17 条，6 大类）

分组汇总：各地区各支付方式销售额

SELECT Region, PaymentMethod,
       SUM(Quantity * UnitPrice * (1 - COALESCE(Discount,0))) AS 销售额,
       COUNT(*) AS 订单数
FROM orders
GROUP BY Region, PaymentMethod
ORDER BY 销售额 DESC;

复购率计算

WITH customer_orders AS (
    SELECT CustomerID, COUNT(DISTINCT InvoiceNo) AS 订单数
    FROM online_retail
    GROUP BY CustomerID
)
SELECT COUNT(*) AS 总客户数,
       SUM(CASE WHEN 订单数 >= 2 THEN 1 ELSE 0 END) AS 复购客户数,
       ROUND(SUM(CASE WHEN 订单数 >= 2 THEN 1 ELSE 0 END)*100.0/COUNT(*),1) AS 复购率
FROM customer_orders;

3. RFM 模型 Python 代码

rfm = df.groupby('CustomerID').agg(
    Recency=('InvoiceDate', lambda x: (today - x.max()).days),
    Frequency=('InvoiceNo', 'nunique'),
    Monetary=('Revenue', 'sum')
)
rfm['R_Score'] = pd.qcut(rfm['Recency'], 4, labels=[4,3,2,1])
rfm['F_Score'] = pd.qcut(rfm['Frequency'].rank(method='first'), 4, labels=[1,2,3,4])
rfm['M_Score'] = pd.qcut(rfm['Monetary'].rank(method='first'), 4, labels=[1,2,3,4])
rfm['RFM_Score'] = rfm['R_Score'] + rfm['F_Score'] + rfm['M_Score']
